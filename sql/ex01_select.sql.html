select 문 
- DML, DQL 
- 사용 빈도가 가장 높다. 
- 데이터 베이스로부터 원하는 데이터를 가져오는 명령어(읽기) 
 
select 문법 
 
[with <sub query>] 
select column_list 
from table_name 
[where search_condition] 
[group by group_by_expression] 
[having search_condition] 
[order by order_expression [asc|desc] 
 
select 컬럼리스트 
from 테이블 
where 절 
group by 절 
having 절 
order by 절 
 
select를 구성하는 모든 절들은 실행 순서가 있다.(***)

-- 테이블 구조?? (스키마)
-- 1. DDL을 본다. (CREATE TABLE tblCountry)
-- 2. 클라이언트 툴 사용한다.

select name, capital, population, continent, area, name
from tblcountry;

select name, '국가명 : ' || name, length(name)
from tblcountry;

select name, capital, population, continent, area
from tblcountry; -- query, sql, 시퀄(sequel)

-- select의 컬럼리스트는 테이블 원본 컬럼 순서와는 무관하다.
-- 비용 높음. 가독성 높음.
-- 가독성이 높기 때문에 선호도가 높음
select area, capital, name, continent, population
from tblcountry;

-- *(all) : 모든 컬럼
-- 비용 낮음. 가독성 낮음
select *
from tblcountry;
