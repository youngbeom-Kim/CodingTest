/* 
 
연산자, Operator 
 
1. 산술 연산자 
- +, -, *, /, % 
 
2. 문자열 연산자 
- 문자열 + 문자열 
- || 
 
3. 비교 연산자 
- >, >=, <, <=, =(==), <>(!=) 
 
4. 논리 연산자 
- and,(&&), or(||), not(!) 
- 컬럼 리스트에서 사용 불가 
- 조건절에서 사용 
 
5. 대입 연산자 
- = 
- 컬럼 = 값 
- (insert), update문에서 사용(추가, 수정) 
- 복합 대인 연산자 없음(+=, -= 등) 
 
6. 3항 연산자 
- 없음 
 
7. 증감 연산자 
- 없음 
 
8. SQL 연산자 
- in, between, like, is, any, all 등... 
 
*/

-- 1. 산술 연산자
select * from tblcomedian;
select first, last, height + 100 from tblcomedian;

-- 컬럼 리스트에 컬럼값이 아닌 상수가 오는 경우...
select first, last, '테스트', 10, now() from tblcomedian;

-- 2. 문자열 연산자
select first, last, last || first from tblcomedian;

-- 3. 비교 연산자
select last || first, weight from tblcomedian where weight > 80;
select last || first, weight, weight > 80 from tblcomedian;
